import "./App.css";
import { useState } from "react"; 

/*
* Model
*/
interface HeaderProps {
	title?: string;
	onChangeMode: () => void;
}

 interface ArticleProps {
	title: string;
	body: string;
}
 

interface TopicProps {
	id: number;
	title: string;
	body: string;
}

interface TopicsProps {
	list: TopicProps[];
	onChangeMode: (id: number) => void;
}

interface CreateProps {
	onCreate: (title: string, body: string) => void;
}

/*
* Component
*/
const Header = (props: HeaderProps) => {
	return (
		<header>
			<h1>
				<a
					href="/"
					onClick={(event) => {
						//새로 로드 되는걸 방지한다
						event.preventDefault();
						//props에서 받아온 이벤트 실행
						props.onChangeMode();
					}}>						
					{props.title}
				</a>
			</h1>
		</header>
	);
};
  
const Article = (props: ArticleProps) => {
	return (
		<article>
			<h2>{props.title}</h2>
			{props.body}
		</article>
	);
};
  

const Nav = (props: TopicsProps) => {
	const lis: JSX.Element[] = [];
	props.list.forEach((topic: TopicProps) => {
		const tag = (
			<li key={topic.id}>
				<a
					id={topic.id.toString()}
					href={"/read/" + topic.id}
					onClick={(event) => {
						//새로 로드 되는걸 방지한다
						event.preventDefault();
						//props에서 받아온 이벤트 실행
						props.onChangeMode(topic.id);
					}}>	
				{topic.title}
				</a>
			</li>
	);
	lis.push(tag);
	}); 

	return (
		<nav>
			<ol>{lis}</ol>
		</nav>
	);
};

  

const Create = (props: CreateProps) => {
	const [title, setTitle] = useState("");
	const [body, setBody] = useState("");

	return (
	<article>
		<h2>Create</h2>
		<form
			onSubmit={(event: React.SyntheticEvent) => {
				console.log("call onSubmit");
				event.preventDefault();
				console.log(`title: ${title}, body: ${body}`);
				props.onCreate(title, body);
		}}>			
			<p>
				<input
					type="text"
					name="title"
					value={title}
					onChange={(event) => setTitle(event.target.value)}
					placeholder="title"
				/>
			</p>
			<p>
				<textarea
					name="body"
					value={body}
					onChange={(event) => setBody(event.target.value)}
					placeholder="body"
>				</textarea>
			</p>
			<p>
				<input type="submit" value="Create" />
			</p>
		</form>
	</article>
	);
};

  

/*
* Main
*/

function App() {
	// const _mode = useState("WELCOME");
	// const mode = _mode[0];
	// const setMode = mode[1];  
	
	let [mode, setMode] = useState("WELCOME");
	let [id, setId] = useState(0);  
	
	//const[value,m setValue] = useState(PRIMITIVE);
	//string, number, boolean, undefined, symblol, null  
	
	//const[value,m setValue] = useState(Object);
	//값을 아예 복사해서 새로 만들어야 함
	// 값이 같으면 새로 안그려서 ref 주소를 변경해야함
	// newValue ={...value}
	
	const [topics, setTopics] = useState<TopicProps[]>([	
		{ id: 0, title: "html1", body: "html1 is..." },	
		{ id: 1, title: "css", body: "css is..." },	
		{ id: 2, title: "js", body: "js is..." },	
	]);

	let content: JSX.Element = <></>; 
	
	if (mode === "WELCOME") {
		console.log("refresh WELCOME");
		content = <Article title="Welcome" body="Hello, WEB"></Article>;
	} else if (mode === "READ") {
		console.log("refresh READ");
		let currentTopic: TopicProps | undefined = topics.find(
		(topic) => topic.id === id);

		if (currentTopic) {
			content = (
				<Article
				title={currentTopic?.title}
				body={currentTopic?.body}>>				
				</Article>			
			);
		} else {
			alert("No id");
		}
	} else if (mode === "CREATE") {
		content = (
		<Create
			onCreate={(title, body) => {
				const newTopic: TopicProps = {
				id: topics.length,
				title: title,
				body: body,
				};
	
				const newTopics: TopicProps[] = [...topics, newTopic];
				setTopics(newTopics);
				setMode("READ");
				setId(newTopic.id);
			}	}
        ></Create>	
	);
}

  

return (
<div className="App">
	<Header
		title="WEB"
		onChangeMode={() => {
		console.log("WELCOME");
		setMode("WELCOME");
	}}>	
	</Header>

  

	<Nav
		list={topics}
		onChangeMode={(_id) => {
		console.log("READ");
		setMode("READ");
		setId(_id);
		}}>
	</Nav>

	{content} 

<ul>
	<li>
		<a
			href="/create"
			onClick={(event) => {
			event.preventDefault();
			setMode("CREATE");
		}}>
		Create
		</a>
	</li>
	
	<li>
		<a
			href="/update"
			onClick={(event) => {
			event.preventDefault();
			setMode("UPDATE");
		}}>
		Update
		</a>
	</li>
</ul>
</div>
);
}
 

export default App;
