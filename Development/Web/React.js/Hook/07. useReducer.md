#### 예제
- state를 업데이트 할때 Reducer를 생성해서 기본적이 사전 작업이나 체크를 하고 업데이트를 한다.

```typescript

//reducer - state를 업데이트 하는 역활(은행)
//dipatch - state업데이트를 요구
//action - 요구의 내용
interface actionType {
  type: string;
  payload: number;
}

const ACTION_TYPE = {
  deposit: "deposit",
  withdraw: "withdraw",
};

const reducer = (state: number, action: actionType): number => {
  console.log("reducer work", state, action);

  switch (action.type) {
    case ACTION_TYPE.deposit:
      return state + action.payload;

    case ACTION_TYPE.withdraw:
      return state - action.payload;

    default:
      return state;
  }
};
function App() {
  const [number, setNumber] = useState(0);
  const [money, dispatch] = useReducer(reducer, 0);

  return (
    <div className="App">
      <h2>useReducer 은행에 오신것을 환영합니다.</h2>
      <p>잔고 : {money}원</p>
      <input
        className="border p-2"
        type="number"
        step="1000"
        value={number}
        onChange={(e: ChangeEvent<HTMLInputElement>) =>
          setNumber(parseInt(e.target.value))
        }
      />
      <button
        onClick={() => {
          dispatch({ type: ACTION_TYPE.deposit, payload: number });
        }}
        className="border p-2"
      >
        예금
      </button>
      <button
        onClick={() => {
          dispatch({ type: ACTION_TYPE.withdraw, payload: number });
        }}
        className="border p-2"
      >
        출금
      </button>
    </div>
  );
}
export default App;

```
